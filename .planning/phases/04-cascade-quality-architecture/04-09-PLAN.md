    ```swift
            let ghostFirstWord = currentGhost.split(whereSeparator: { $0.isWhitespace }).first.map(String.init) ?? ""
            guard !typedFirstWord.isEmpty, !ghostFirstWord.isEmpty else { return nil }
            let ghostStartsWithTyped = ghostFirstWord.lowercased().hasPrefix(typedFirstWord.lowercased())
            if ghostStartsWithTyped { return nil }
            if timeSinceShowMs <= SuggestionPolicy.Tuning.badMaxDivergeMs {
                return .typedDiverged
            }
            if timeSinceShowMs >= SuggestionPolicy.Tuning.uselessMinVisibleMs {
                return .typedPastWithoutOverlap
            }
            return nil
        }
    }
    ```
