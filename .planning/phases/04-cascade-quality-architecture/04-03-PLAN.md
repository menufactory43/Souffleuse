struct GenerationToken: Sendable, Equatable {
    let value: UInt64
}

@MainActor
final class GenerationPlanner {
    private(set) var currentGeneration: GenerationToken
    private var currentTask: Task<Void, Never>?
    private var debounceTask: Task<Void, Never>?
    static let predictDebounceNanos: UInt64 = 30 * 1_000_000  // 30ms

    init()

    /// Cancels in-flight Task and bumps generation. Returns the new token to capture in the new request.
    func beginGeneration() -> GenerationToken

    /// True if `token` is still the active generation. Use in stream chunk closures
    /// to drop stale chunks deterministically.
    func isCurrent(_ token: GenerationToken) -> Bool

    /// Sets the current Task — the Task created by the caller for the request lifecycle.
    func setCurrentTask(_ task: Task<Void, Never>)

    /// Cancels the current Task + bumps generation. Idempotent.
    func cancel()

    /// Debounce a piece of work by predictDebounceNanos. Cancels any pending debounce.
    func scheduleDebounced(_ work: @escaping @Sendable @MainActor () async -> Void)
}
