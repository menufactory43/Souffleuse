    maxTokens: maxTokens,
    temperature: 0,
    topP: 0.9,
    repetitionPenalty: 1.0,
    repetitionContextSize: 32

let envBypass = ProcessInfo.processInfo.environment["SOUFFLEUSE_DISABLE_KV_CACHE"]?.isEmpty == false
// Canonicalise the few-shot block BEFORE fingerprinting (Warning #2 from
// plan-checker iteration 2): SimilarHistoryRetrieval returns entries in
// similarity-ranked order, which is not stable across predicts. Feeding the
// raw block here would flip the fingerprint on every predict and silently
// invalidate the cache → defeats the entire phase win. The canonical form is
// only used for the fingerprint; the actual `examplesBlock` fed to the LLM
// keeps its similarity-ranked order (it stays in `promptTokens`).
let canonicalExamplesBlock = InvariancePrefix.canonicalizePreviousUserInputs(examplesBlock)
let invariance = InvariancePrefix(
    system: baseSystem,
    customInstructions: customInstr,
    contextPrefix: ctxPrefix,
    fieldContext: fieldContextSlot,
    afterCursor: afterCursorSlot,
    previousUserInputs: canonicalExamplesBlock
)
let newFingerprint = invariance.fingerprint

let userTailTokenCount = tokenCountCache.count(of: userTail, with: context.tokenizer)
let invariantPrefixTokenCount = max(0, promptTokens.count - userTailTokenCount)

var chosenCache: [KVCache]
var iteratorInputTokens: [Int] = promptTokens  // default: feed full prompt
if envBypass {
    chosenCache = makePromptCache(model: context.model, parameters: params)
} else if sessionCacheHolder.caches == nil {
    let fresh = makePromptCache(model: context.model, parameters: params)
    chosenCache = fresh
    sessionCacheHolder.install(
        caches: fresh,
        fingerprint: newFingerprint,
        beforeCursorTokens: userTailTokenCount
    )
    Log.info(.predictor, "kv_cache_invalidate", count: 0) // .cold
} else if sessionCacheHolder.fingerprint != newFingerprint {
    sessionCacheHolder.invalidate(reason: .fingerprintChanged)
    let fresh = makePromptCache(model: context.model, parameters: params)
    chosenCache = fresh
    sessionCacheHolder.install(
        caches: fresh,
        fingerprint: newFingerprint,
        beforeCursorTokens: userTailTokenCount
    )
    Log.info(.predictor, "kv_cache_invalidate", count: 1) // .fingerprintChanged
} else {
    let existing = sessionCacheHolder.caches!
    let oldCount = sessionCacheHolder.beforeCursorTokens
    let newCount = userTailTokenCount
    chosenCache = existing
    if newCount > oldCount {
        let beforeCursorTokens = Array(promptTokens.suffix(newCount))
        let deltaTokens = Array(beforeCursorTokens.dropFirst(oldCount))
        iteratorInputTokens = deltaTokens
        sessionCacheHolder.updateBeforeCursorTokens(newCount)
        Log.info(.predictor, "kv_cache_extend", count: newCount - oldCount)
    } else if newCount < oldCount, canTrimPromptCache(existing) {
        let diff = oldCount - newCount
        _ = trimPromptCache(existing, numTokens: diff)
        sessionCacheHolder.updateBeforeCursorTokens(newCount)
        iteratorInputTokens = []  // cache now represents exact new prefix
        Log.info(.predictor, "kv_cache_trim", count: diff)
    } else if newCount < oldCount {
        sessionCacheHolder.invalidate(reason: .beforeCursorDiverged)
        let fresh = makePromptCache(model: context.model, parameters: params)
        chosenCache = fresh
        sessionCacheHolder.install(
            caches: fresh, fingerprint: newFingerprint, beforeCursorTokens: newCount
        )
        iteratorInputTokens = promptTokens  // full prefill
        Log.info(.predictor, "kv_cache_invalidate", count: 2) // .beforeCursorDiverged
    }
}

let input = LMInput(tokens: MLXArray(iteratorInputTokens))

let snapshot: NgramSnapshot? = personalizationStrength > 0
    ? await ngramModel.snapshot()
    : nil
let stream: AsyncStream<Generation>
if let snapshot, !snapshot.isEmpty, personalizationStrength > 0 {
    let repetition = RepetitionContext(repetitionPenalty: 1.0, repetitionContextSize: 32)
    let bias = NgramLogitBias(snapshot: snapshot, strength: personalizationStrength)
    let chain = ChainLogitProcessor(first: repetition, second: bias)
    let iterator = try TokenIterator(
        input: input,
        model: context.model,
        cache: chosenCache,
        processor: chain,
        sampler: params.sampler(),
        maxTokens: maxTokens
    )
    stream = MLXLMCommon.generate(input: input, context: context, iterator: iterator)
} else {
    let iterator = try TokenIterator(
        input: input,
        model: context.model,
        cache: chosenCache,
        processor: nil,
        sampler: params.sampler(),
        maxTokens: maxTokens
    )
    stream = MLXLMCommon.generate(input: input, context: context, iterator: iterator)
